# Nyan Cat Theme Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a compiled-in Nyan Cat theme with an animated GIF cat sprite, wave-distorted rainbow trail, and pixel-art twinkling stars — all assets embedded in the DLL at build time.

**Architecture:** A CMake custom command splits `nyan-cat.gif` into 16 PNGs and generates `src/nyancat_frames.h` (16 byte arrays + lookup tables) at build time. At runtime, `AddonLoad` loads all 16 frames as Nexus textures into `s_NyanFrames[16]`. Two draw hooks (`NyanCatDrawChatBg`, `NyanCatDrawContactsBg`) and `BuildNyanCatTheme()` follow the established pattern used by Sylvari Grove and other compiled-in themes.

**Tech Stack:** C++17 / Dear ImGui / Win32 / Nexus addon API. Cross-compiled on Linux with MinGW. Build: `cd build && make`. No test suite — verification is a clean build.

- [ ] **Step 1: Create the frame-generation shell script**

Create `cmake/gen_nyan_frames.sh`:

```bash
#!/usr/bin/env bash
# Usage: gen_nyan_frames.sh <gif> <builddir> <outfile>
set -e
GIF="$1"
BUILDDIR="$2"
OUTFILE="$3"

# Split GIF into 16 coalesced frames
convert "$GIF" -coalesce +adjoin "$BUILDDIR/nyan_frame_%02d.png"

{
    echo "#pragma once"
    echo "// Auto-generated by cmake/gen_nyan_frames.sh -- do not edit"
    echo ""

    for i in $(seq -w 0 15); do
        xxd -i "$BUILDDIR/nyan_frame_${i}.png" \
            | sed "s/unsigned char .*\[\]/static const unsigned char kNyanFrame${i}[]/" \
            | sed "s/unsigned int .*_len/static const unsigned int kNyanFrame${i}Len/"
        echo ""
    done

    echo "static const unsigned char* const kNyanFrameData[16] = {"
    for i in $(seq -w 0 15); do
        echo "    kNyanFrame${i},"
    done
    echo "};"
    echo ""
    echo "static const unsigned int kNyanFrameSize[16] = {"
    for i in $(seq -w 0 15); do
        echo "    kNyanFrame${i}Len,"
    done
    echo "};"
} > "$OUTFILE"
```

## Task 3: Pixel-Art Star Helper

**Files:**
- Modify: `src/dllmain.cpp` (new static function, placed near other draw hooks around line 2138)

Each star cycles through five stages over 2.5 s. All drawing uses `AddRectFilled` (1×1 or 2×2 pixel blocks). Position is deterministic — no `rand()`, no heap allocation.

- [ ] **Step 1: Add NyanCatStars function**

Add this function immediately before the `// ── Sylvari Grove draw hooks` comment (around line 2138):

```cpp
// ── Nyan Cat draw hooks ───────────────────────────────────────────────────────

static void NyanCatStars(ImDrawList* dl, ImVec2 mn, ImVec2 mx, int count) {
    float t = (float)ImGui::GetTime();
    float w = mx.x - mn.x;
    float h = mx.y - mn.y;
    constexpr float kCycle  = 2.5f;
    constexpr ImU32 kWhite  = IM_COL32(255, 255, 255, 220);
    constexpr float kAngles[8] = {
        0.000f, 0.785f, 1.571f, 2.356f,
        3.142f, 3.927f, 4.712f, 5.498f
    };

    for (int i = 0; i < count; i++) {
        float phi   = i * 0.6180339f;   // golden ratio
        float psi   = i * 0.3819660f;
        float phase = i * (kCycle / count);
        float frac  = fmodf(t + phase, kCycle) / kCycle;  // 0..1 within cycle

        // Rerandomise position each complete cycle
        int   seed = (int)((t + phase) / kCycle);
        float sx   = mn.x + fmodf(phi * (float)(seed * 7 + 3), 1.0f) * w;
        float sy   = mn.y + fmodf(psi * (float)(seed * 13 + 5), 1.0f) * h;

        if (frac < 0.15f) {
            // Stage 1: 2×2 dot
            dl->AddRectFilled(ImVec2(sx, sy), ImVec2(sx + 2, sy + 2), kWhite);
        } else if (frac < 0.35f) {
            // Stage 2: small cross — 3×3 centre + 1 px arms (7 px span)
            dl->AddRectFilled(ImVec2(sx - 1, sy - 1), ImVec2(sx + 2, sy + 2), kWhite);
            dl->AddRectFilled(ImVec2(sx,     sy - 2), ImVec2(sx + 1, sy - 1), kWhite);
            dl->AddRectFilled(ImVec2(sx,     sy + 2), ImVec2(sx + 1, sy + 3), kWhite);
            dl->AddRectFilled(ImVec2(sx - 2, sy    ), ImVec2(sx - 1, sy + 1), kWhite);
            dl->AddRectFilled(ImVec2(sx + 2, sy    ), ImVec2(sx + 3, sy + 1), kWhite);
        } else if (frac < 0.60f) {
            // Stage 3: large cross — 3×3 centre + 4 px arms (11 px span)
            dl->AddRectFilled(ImVec2(sx - 1, sy - 1), ImVec2(sx + 2, sy + 2), kWhite);
            for (int a = 2; a <= 5; a++) {
                dl->AddRectFilled(ImVec2(sx,       sy - a    ), ImVec2(sx + 1, sy - a + 1), kWhite);
                dl->AddRectFilled(ImVec2(sx,       sy + a - 1), ImVec2(sx + 1, sy + a    ), kWhite);
                dl->AddRectFilled(ImVec2(sx - a,   sy        ), ImVec2(sx - a + 1, sy + 1), kWhite);
                dl->AddRectFilled(ImVec2(sx + a - 1, sy      ), ImVec2(sx + a,     sy + 1), kWhite);
            }
        } else if (frac < 0.80f) {
            // Stage 4: pixel ring — 8 single-pixel dots at radius 4
            for (float ang : kAngles) {
                float rx = sx + cosf(ang) * 4.0f;
                float ry = sy + sinf(ang) * 4.0f;
                dl->AddRectFilled(ImVec2(rx, ry), ImVec2(rx + 1, ry + 1), kWhite);
            }
        }
        // Stage 5 (frac 0.80–1.0): invisible, position reseeds next cycle
    }
}
```

## Task 4: Rainbow Trail + Chat/Contacts Draw Hooks

**Files:**
- Modify: `src/dllmain.cpp` (three new static functions, immediately after `NyanCatStars`)

The rainbow is drawn as 3 px-wide vertical strips from the left edge of the pane to the cat's left edge. A sine wave distorts each strip's Y position, producing a wave that travels right-to-left. Alpha fades from 0.30 at the left to 0.95 at the cat. The chat hook also renders the current GIF frame (UV-cropped to skip the built-in rainbow stub). The contacts hook renders stars only.

- [ ] **Step 1: Add NyanCatRainbow function**

Immediately after `NyanCatStars`, add:

```cpp
static void NyanCatRainbow(ImDrawList* dl, ImVec2 mn,
                            float cat_left_x, float cat_center_y) {
    static const ImU32 kBands[6] = {
        IM_COL32(255,  30,   0, 255),
        IM_COL32(255, 153,   0, 255),
        IM_COL32(255, 240,   0, 255),
        IM_COL32( 51, 210,   0, 255),
        IM_COL32( 51, 102, 255, 255),
        IM_COL32(160,   0, 255, 255),
    };
    constexpr float kBandH      = 8.0f;
    constexpr float kWaveAmp    = 6.0f;
    constexpr float kWaveLen    = 180.0f;
    constexpr float kWavePeriod = 0.8f;
    constexpr float kStripW     = 3.0f;

    float t         = (float)ImGui::GetTime();
    float k         = 2.0f * 3.14159265f / kWaveLen;
    float omega     = 2.0f * 3.14159265f / kWavePeriod;
    float rainbowTop = cat_center_y - 3.0f * kBandH;  // 3 bands above centre
    float span      = cat_left_x - mn.x;
    if (span <= 0.0f) return;

    for (float x = mn.x; x < cat_left_x; x += kStripW) {
        float wave    = kWaveAmp * sinf(k * (x - mn.x) + omega * t);
        float progress = (x - mn.x) / span;
        int   ia      = (int)((0.30f + progress * 0.65f) * 255.0f);

        for (int b = 0; b < 6; b++) {
            ImU32 col = (kBands[b] & 0x00FFFFFFu) | ((ImU32)ia << 24);
            float y0  = rainbowTop + wave + b * kBandH;
            dl->AddRectFilled(ImVec2(x, y0),
                              ImVec2(x + kStripW + 1.0f, y0 + kBandH + 1.0f), col);
        }
    }
}
```

```cpp
static void NyanCatDrawChatBg(ImDrawList* dl, ImVec2 mn, ImVec2 mx) {
    float w = mx.x - mn.x;
    float h = mx.y - mn.y;

    // Cat sprite: maintain 498:280 aspect ratio, 38% of pane height
    float cat_h     = h * 0.38f;
    float cat_w     = cat_h * (498.0f / 280.0f);
    float cat_left  = mn.x + w * 0.50f;          // centred horizontally
    float cat_top   = mn.y + h * 0.48f - cat_h * 0.5f;

    // Rainbow connects at ~45% down the sprite (centre of rainbow bands)
    float cat_center_y = cat_top + cat_h * 0.45f;

    NyanCatRainbow(dl, mn, cat_left, cat_center_y);

    // Cycle GIF frame at 50 ms per frame (20 fps)
    int       frame = (int)((float)ImGui::GetTime() / 0.05f) % 16;
    Texture_t* tex  = s_NyanFrames[frame];
    if (tex && tex->Resource) {
        // UV x=0.25 crops the built-in rainbow stub; only the cat body is drawn
        dl->AddImage(
            (ImTextureID)tex->Resource,
            ImVec2(cat_left, cat_top),
            ImVec2(cat_left + cat_w, cat_top + cat_h),
            ImVec2(0.25f, 0.0f),
            ImVec2(1.0f,  1.0f));
    }

    NyanCatStars(dl, mn, mx, 20);
}
```
